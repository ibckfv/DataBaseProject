## Дедлок

```sql
-- Сессия 1
BEGIN;
UPDATE clients SET name = 'Deadlock 1' WHERE client_id = 100;
```

```sql
-- Сессия 2
BEGIN;
UPDATE clients SET name = 'Deadlock 2' WHERE client_id = 101;
```

```sql
-- Сессия 1 теперь пытается обновить то, что заблокировала сессия 2
```

## Режимы блокировки на уровне строк

### FOR UPDATE

```sql
-- Сессия 1
BEGIN;
SELECT * FROM orders WHERE order_id = 1 FOR UPDATE;

-- Сессия 2
BEGIN;
SELECT * FROM orders WHERE id = 1 FOR UPDATE;
-- Ждет (заблокировано сессией 1)

-- Сессия 1
COMMIT;  -- Освобождает блокировку

-- Сессия 2 теперь получает блокировку и завершает
COMMIT;
```

### FOR NO KEY UPDATE

```sql
-- Сессия 1
BEGIN;
SELECT * FROM orders WHERE order_id = 1 FOR NO KEY UPDATE;

-- Сессия 2
BEGIN;
SELECT * FROM orders WHERE order_id = 1 FOR UPDATE;
-- Ждет, FOR UPDATE конфликтует с FOR NO KEY UPDATE
```

### FOR SHARE

```sql
-- Сессия 1
BEGIN;
SELECT * FROM orders WHERE order_id = 1 FOR SHARE;

-- Сессия 2
BEGIN;
SELECT * FROM orders WHERE order_id = 1 FOR UPDATE;
-- Ждет, FOR SHARE не блокирует чтение, но блокирует изменение

-- Сессия 3 может читать свободно
SELECT * FROM orders WHERE order_id = 1;  -- Не ждет
```

### FOR KEY SHARE

```sql
-- Сессия 1
BEGIN;
SELECT * FROM orders WHERE order_id = 1 FOR KEY SHARE;

-- Сессия 2
BEGIN;
UPDATE orders SET data = 'new' WHERE order_id = 1;
-- Не ждет, FOR KEY SHARE блокирует только удаление, изменение ключей
COMMIT;
```
